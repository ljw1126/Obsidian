    public int[][] solution(int[][] arr1, int[][] arr2) {

class Solution {
    public int solution(int n) {
        int answer = 0;
        int sum = 0;
        int R = 0;
        for(int L = 1; L <= n; L++) {
            
            while(R + 1 <= n && sum < n) {
                R += 1;
                sum += R;            
            }
            
            if(sum == n) answer += 1;
            
            sum -= L;
        }
        
        return answer;
    }
}
