### 문제

>[!example]
>입력 예시
>7 8
>0 1 3
>1 1 7
>0 7 6
>1 7 1
>0 3 7
>0 4 2
>0 1 1
>1 1 1
>
>출력 예시
>NO
>NO
>YES
### 풀이

```java
import java.util.*;
import java.io.*;

public class Main {
	private static StringBuilder sb = new StringBuilder();  
	private static InputProcessor inputProcessor = new InputProcessor();  
	  
	public static void main(String[] args) {  
	    input();  
	    pro();  
	    output();  
	}  
	  
	private static int n, m;  
	  
	private static void input() {  
	    n = inputProcessor.nextInt(); // 0 ~ N 번 팀이 있다  
	    m = inputProcessor.nextInt(); // 연산 횟수  
	}  
	  
	private static void pro() {  
	    int[] parent = new int[n + 1];  
	    for(int i = 0; i <= n; i++) {  
	        parent[i] = i;  
	    }  
	  
	    for(int i = 1; i <= m; i++) {  
	        // 0 : 합치기, 1 : 같은 팀 여부 확인  
	        int cmd = inputProcessor.nextInt(); 
	        int a = inputProcessor.nextInt();  
	        int b = inputProcessor.nextInt();  
	  
	        if(cmd == 0) union(parent, a, b);  
	        else sb.append(isTeammate(a, b, parent) ? "YES" : "NO").append("\n");  
	    }  
	}  
	  
	private static void union(int[] parent, int a, int b) {  
	    int _a = find(parent, a);  
	    int _b = find(parent, b);  
	  
	    if(_a < _b) {  
	        parent[_b] = _a;  
	    } else {  
	        parent[_a] = _b;  
	    }  
	}  
	  
	private static int find(int[] parent, int n) {  
	    if(n == parent[n]) return n;  
	  
	    return parent[n] = find(parent, parent[n]);  
	}  
	  
	private static boolean isTeammate(int a, int b, int[] parent) {  
	    return find(parent, a) == find(parent, b);  
	}  
	  
	private static void output() {  
	    try (BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out))) {  
	        bw.write(sb.toString());  
	        bw.flush();  
	    } catch (IOException e) {  
	        throw new RuntimeException(e);  
	    }  
	}  
	  
	private static class InputProcessor {  
	    private BufferedReader br;  
	    private StringTokenizer st;  
	  
	    public InputProcessor() {  
	        this.br = new BufferedReader(new InputStreamReader(System.in));  
	    }  
	  
	    public String next() {  
	        while (st == null || !st.hasMoreElements()) {  
	            try {  
	                st = new StringTokenizer(br.readLine());  
	            } catch (IOException e) {  
	                throw new RuntimeException(e);  
	            }  
	        }  
	  
	        return st.nextToken();  
	    }  
	  
	    public String nextLine() {  
	        String result = "";  
	  
	        try {  
	            result = br.readLine();  
	        } catch (IOException e) {  
	            throw new RuntimeException(e);  
	        }  
	  
	        return result;  
	    }  
	  
	    public int nextInt() {  
	        return Integer.parseInt(next());  
	    }  
	  
	    public long nextLong() {  
	        return Long.parseLong(next());  
	    }  
	}
}
```
