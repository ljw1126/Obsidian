### 문제 링크
- USACO

### 시간 복잡도 
- `O(ElogV)`
	- V : 노드 수, 최대 20,000개
	- E : 간선 수, 50,000개 (양방향이므로 * 2)
	- `O(100,000 log 20,000)`

### 풀이
- 다익스트라 알고리즘 사용
- `{java}int[] dist` 를 최대값으로 초기화 후 1번 인덱스부터 최단 경로 탐색 
-  **최단 거리가 가장 먼 헛간**이 가장 안전하다고 판단하고 있습니다
	- 이때 최단 거리의 의미는 지나야 하는 길의 최소 개수를 의미
	- <u>최소 거리가 가장 큰 위치를 찾으면 된다</u>

입출력 예시 
```text
입력
6 7
3 6
4 3
3 2
1 3
1 2
2 4
5 2

출력
4 2 3
```

### 제출 코드
```java hl:50,53
import java.util.*;
import java.io.*;

public class Main {
	
	private static StringBuilder sb = new StringBuilder();  
	private static InputProcessor inputProcessor = new InputProcessor();  
	  
	public static void main(String[] args) {  
	    input();  
	    pro();  
	    output();  
	}  
	  
	private static final int INF = 100_000_001;  
	private static int n, m;  
	private static List<List<Integer>> adj;  
	  
	private static void input() {  
	    n = inputProcessor.nextInt(); // 헛간 수(노드)  
	    m = inputProcessor.nextInt(); // m개의 양방향 간선  
	  
	    adj = new ArrayList<>();  
	    for(int i = 0; i <= n; i++) {  
	        adj.add(new ArrayList<>());  
	    }  
	  
	    for(int i = 1; i <= m; i++) {  
	        int a = inputProcessor.nextInt();  
	        int b = inputProcessor.nextInt();  
	  
	        adj.get(a).add(b);  
	        adj.get(b).add(a);  
	    }  
	}  
	  
	private static void pro() {  
	    Queue<Node> pq = new PriorityQueue<>();  
	    pq.add(new Node(1, 0));  
	  
	    int[] dist = new int[n + 1];  
	    Arrays.fill(dist, INF);  
	    dist[1] = 0;  
	  
	    while(!pq.isEmpty()) {  
	        Node cur = pq.poll();  
	        int no = cur.no;  
	        int _dist = cur.dist;  
	  
	        if(dist[no] < _dist) continue;  
	  
	        for(int next : adj.get(no)) {  
	            if(dist[next] > _dist + 1) {  
	                dist[next] = _dist + 1;  
	                pq.add(new Node(next, dist[next]));  
	            }  
	        }  
	    }  
	  
	    int no = 0;  
	    int maxDist = 0;  
	    int count = 0;  
	    for(int i = 1; i <= n; i++) {  
	        if(maxDist < dist[i]) {  
	            maxDist = dist[i];  
	            no = i;  
	            count = 1;  
	        } else if(maxDist == dist[i]) {  
	            count += 1;  
	        }  
	    }  
	  
	    sb.append(no).append(" ")  
	            .append(maxDist).append(" ")  
	            .append(count);  
	}  
	  
	private static class Node implements Comparable<Node> {  
	    private final int no;  
	    private final int dist;  
	  
	    public Node(int no, int dist) {  
	        this.no = no;  
	        this.dist = dist;  
	    }  
	  
	    public int compareTo(Node o) {  
	        return this.dist - o.dist;  
	    }  
	}  
	  
	private static void output() {  
	    try (BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out))) {  
	        bw.write(sb.toString());  
	        bw.flush();  
	    } catch (IOException e) {  
	        throw new RuntimeException(e);  
	    }  
	}  
	  
	private static class InputProcessor {  
	    private BufferedReader br;  
	    private StringTokenizer st;  
	  
	    public InputProcessor() {  
	        this.br = new BufferedReader(new InputStreamReader(System.in));  
	    }  
	  
	    public String next() {  
	        while (st == null || !st.hasMoreElements()) {  
	            try {  
	                st = new StringTokenizer(br.readLine());  
	            } catch (IOException e) {  
	                throw new RuntimeException(e);  
	            }  
	        }  
	  
	        return st.nextToken();  
	    }  
	  
	    public String nextLine() {  
	        String result = "";  
	  
	        try {  
	            result = br.readLine();  
	        } catch (IOException e) {  
	            throw new RuntimeException(e);  
	        }  
	  
	        return result;  
	    }  
	  
	    public int nextInt() {  
	        return Integer.parseInt(next());  
	    }  
	  
	    public long nextLong() {  
	        return Long.parseLong(next());  
	    }  
	}

}
```
