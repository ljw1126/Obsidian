https://www.acmicpc.net/problem/11404

- 시간복잡도 : O(N^3)
- 추가문제 
	- https://www.acmicpc.net/problem/13141


```java
import java.util.*;
import java.io.*;

public class Main {
   private static StringBuilder sb = new StringBuilder();

    public static void main(String[] args) throws IOException {
        input();
        pro();
        output();
    }

    private static final int MAX_VALUE = 10000001;
    private static int n, m;
    private static int[][] floyed;

    private static void input() throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

        n = Integer.parseInt(br.readLine());
        m = Integer.parseInt(br.readLine());

        floyed = new int[n + 1][n + 1];
        for(int i = 1; i <= n; i++) {
            Arrays.fill(floyed[i], MAX_VALUE);
            floyed[i][i] = 0;
        }

        for(int i = 0; i < m; i++) {
            StringTokenizer st = new StringTokenizer(br.readLine());
            int a = Integer.parseInt(st.nextToken());
            int b = Integer.parseInt(st.nextToken());
            int c = Integer.parseInt(st.nextToken());

            floyed[a][b] = Math.min(floyed[a][b], c);
        }
    }

    private static void pro() {
        for(int k = 1; k <= n; k++) {
            for(int i = 1; i <= n; i++) {
                for(int j = 1; j <= n; j++) {
                    if(i == j) continue;

                    floyed[i][j] = Math.min(floyed[i][j], floyed[i][k] + floyed[k][j]);
                }
            }
        }

        for(int i = 1; i <= n; i++) {
            for(int j = 1; j <= n; j++) {
                if(floyed[i][j] == MAX_VALUE) sb.append(0).append(" ");
                else sb.append(floyed[i][j]).append(" ");
            }

            sb.append("\n");
        }
    }

    private static void output() throws IOException {
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));
        bw.write(sb.toString());
        bw.flush();
        bw.close();
    }
    
}
```
