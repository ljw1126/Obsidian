```java
import java.util.*;
import java.io.*;

public class Main {
    private static StringBuilder sb = new StringBuilder();

    public static void main(String[] args) throws IOException {
        input();
        pro();
        output();
    }

    private static int n, k;

    private static void input() throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer st = new StringTokenizer(br.readLine());

        n = Integer.parseInt(st.nextToken()); // 수빈이가 있는 위치(시작점)
        k = Integer.parseInt(st.nextToken()); // 동생이 있는 위치 (도착점)
    }

    private static void pro() {
        Queue<Node> pq = new PriorityQueue<>();
        pq.add(new Node(n, 0));

        int[] dist = new int[100001];
        Arrays.fill(dist, Integer.MAX_VALUE);
        dist[n] = 0;

        int result = -1;
        while(!pq.isEmpty()) {
            Node cur = pq.poll();

            if(dist[cur.idx] < cur.dist) continue;
            if(cur.idx == k) {
                result = cur.dist;
                break;
            }
            
            // 워프
            if(inRange(cur.idx * 2) && dist[cur.idx * 2] > dist[cur.idx]) {
                dist[cur.idx * 2] = dist[cur.idx];
                pq.add(new Node(cur.idx * 2, dist[cur.idx * 2]));
            }

						// +1, -1의 경우
            if(inRange(cur.idx + 1) && dist[cur.idx + 1] > dist[cur.idx] + 1) {
                dist[cur.idx + 1] = dist[cur.idx] + 1;
                pq.add(new Node(cur.idx + 1, dist[cur.idx + 1]));
            }

            if(inRange(cur.idx - 1) && dist[cur.idx - 1] > dist[cur.idx] + 1) {
                dist[cur.idx - 1] = dist[cur.idx] + 1;
                pq.add(new Node(cur.idx - 1, dist[cur.idx - 1]));
            }
        }

        sb.append(result);
    }

    private static boolean inRange(int idx) {
        return idx >= 0 && idx <= 100000;
    }

    private static class Node implements Comparable<Node> {
        private int idx;
        private int dist;

        public Node(int idx, int dist) {
            this.idx = idx;
            this.dist = dist;
        }

        public int compareTo(Node o) {
            return this.dist - o.dist;
        }
    }

    private static void output() throws IOException {
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));
        bw.write(sb.toString());
        bw.flush();
        bw.close();
    }
}
```
