https://www.acmicpc.net/problem/1976

- union-find 알고리즘 활용
- 간선 정보를 수집 후 순차 조회하면서 union-find 처리
- 이후 여행 일정 경로를 확인시 각 노드가 같은 집합(`int[] parents`)에 소속되어 있다면 여행가다는 의미이고, 다른 부모를 가진다면 서로 다른 집합이라 여행 불가하다는 의미에 해당

```java
import java.util.*;
import java.io.*;

public class Main {
    
    private static final StringBuilder sb = new StringBuilder();

    public static void main(String[] args) throws IOException {
        input();
        pro();
        output();
    }

    private static int n, m;
    private static List<Edge> edges;
    private static int[] travelPlan;

    private static void input() throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

        n = Integer.parseInt(br.readLine()); // 도시의 수 (노드)
        m = Integer.parseInt(br.readLine()); // 여행 계획에 속한 도시들의 수

        edges = new ArrayList<>();
        for (int i = 1; i <= n; i++) {
            StringTokenizer st = new StringTokenizer(br.readLine());
            for (int j = 1; j <= n; j++) {
                int v = Integer.parseInt(st.nextToken());

                if (v == 0) continue;

                edges.add(new Edge(i, j));
            }
        }

        travelPlan = new int[m];
        StringTokenizer st = new StringTokenizer(br.readLine());
        for (int i = 0; i < m; i++) {
            travelPlan[i] = Integer.parseInt(st.nextToken());
        }
    }

    private static class Edge {
        private int from;
        private int to;

        public Edge(int from, int to) {
            this.from = from;
            this.to = to;
        }
    }

    private static void pro() {
				// union-find 
        int[] parents = new int[n + 1];
        for (int i = 1; i <= n; i++) {
            parents[i] = i;
        }

        for (Edge e : edges) {
            if (!isSameParent(parents, e.from, e.to)) {
                unionFind(parents, e.from, e.to);
            }
        }

				//(union-find 결과 활용해) 같은 집합인 경우 여행이 가능하다는 의미
        boolean isPossible = true;
        for (int i = 0; i < m - 1; i++) {
            if (parents[travelPlan[i]] == parents[travelPlan[i + 1]]) {
                continue;
            }

            isPossible = false;
            break;
        }

        sb.append(isPossible ? "YES" : "NO");
    }

    private static void unionFind(int[] parents, int from, int to) {
        int a = findParent(parents, from);
        int b = findParent(parents, to);

        if (a < b) {
            parents[b] = a;
        } else {
            parents[a] = b;
        }
    }

    private static boolean isSameParent(int[] parents, int from, int to) {
        return findParent(parents, from) == findParent(parents, to);
    }

    private static int findParent(int[] parents, int node) {
        if (parents[node] == node) return node;

        return parents[node] = findParent(parents, parents[node]);
    }

    private static void output() throws IOException {
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));
        bw.write(sb.toString());
        bw.flush();
        bw.close();
    }
    
}
```
