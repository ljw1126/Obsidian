https://www.acmicpc.net/problem/1463

- 처음 재귀 방식으로 풀이 시도를 했고, StackOverflow를 해결하지 못함
    - x 의 최대가 **10^6**인데, 재귀가 너무 깊어 overflow 발생한 것으로 파악
        - 파이썬의 경우 **1,000번**, 자바의 경우 **10,000번** 재귀 호출부터 `스택오버플로우` 발생한다는 걸 알게 됨
    - 종료조건, 메모리제이션은 했는데 .. **가지치기**를 누락해서 실패한거라 생각(250710)
- **bottom-up** 방식으로 풀이하여 통과

```java
import java.util.*;
import java.io.*;

public class Main {
    private static StringBuilder sb = new StringBuilder();

    public static void main(String[] args) throws IOException {
        input();
        pro();
        output();
    }
    
    private static int x;

    private static void input() throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        x = Integer.parseInt(br.readLine());
    }

		// 1에서 x가 되기 위한 최소 연산 횟수를 구한다
    private static void pro() {
        int[] dp = new int[1000001];
        Arrays.fill(dp, Integer.MAX_VALUE);
        dp[1] = 0;

        for(int i = 2; i <= x; i++) {
            dp[i] = dp[i - 1] + 1;

            if(i % 2 == 0) {
                dp[i] = Math.min(dp[i], dp[i / 2] + 1);
            }
            
            if(i % 3 == 0) {
                dp[i] = Math.min(dp[i], dp[i / 3] + 1);
            }
        }

        sb.append(dp[x]);
    }

    private static void output() throws IOException {
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));
        bw.write(sb.toString());
        bw.flush();
        bw.close();
    }

    
}
```


top-down 풀이시 먼저 3,2 연산을 한 후에 -1을 해야 스택오버플로우가 안나는 거였음
```java
import java.util.*;
import java.io.*;

public class Main {
    
    private static StringBuilder sb = new StringBuilder();

    public static void main(String[] args) throws IOException {
        input();
        pro();
        output();
    }

    private static final int MAX_VALUE = 1000001;
    private static int n;

    private static void input() throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        n = Integer.parseInt(br.readLine());
    }

    private static void pro() {
        int[] dp = new int[1000001];
        Arrays.fill(dp, 1000001);

        dp[0] = 0;
        dp[1] = 0;
        dp[2] = 1;
        dp[3] = 1;

        topDown(n, dp);

        sb.append(dp[n]);
    }

    private static int topDown(int x, int[] dp) {
        if(dp[x] != MAX_VALUE) {
            return dp[x];
        }

        int result = MAX_VALUE;
        if(x % 3 == 0) {
            result = Math.min(result, topDown(x / 3, dp) + 1);
        }

        if(x % 2 == 0) {
            result = Math.min(result, topDown(x / 2, dp) + 1);
        }
        
        result = Math.min(result, topDown(x - 1, dp) + 1);

        return dp[x] = result;
    }

    private static void output() throws IOException {
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));
        bw.write(sb.toString());
        bw.flush();
        bw.close();
    }

    
}
```
