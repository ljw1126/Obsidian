https://www.acmicpc.net/problem/1068

- 루트를 삭제하는 경우 리프 노드는 0이 되므로, 예외 케이스만 조건문 처리

```java
import java.util.*;
import java.io.*;

public class Main {
     private static StringBuilder sb = new StringBuilder();

    public static void main(String[] args) throws IOException {
        input();
        pro();
        output();
    }

    private static int n, target, root;
    private static List<List<Integer>> adj;

    private static void input() throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

        n = Integer.parseInt(br.readLine()); // 트리의 노드 개수

        adj = new ArrayList<>();
        for(int i = 0; i <= n; i++) {
            adj.add(new ArrayList<>());
        }

        StringTokenizer st = new StringTokenizer(br.readLine());
        for(int i = 0; i < n; i++) {
            int parent = Integer.parseInt(st.nextToken());
            if(parent == -1) {
                root = i;
                continue;
            }

            adj.get(parent).add(i);
        }


        target = Integer.parseInt(br.readLine()); // 삭제할 번호
    }

    private static void pro() {
        if(target == root) {
            sb.append(0);
            return;
        }

        for(int i = 0; i < n; i++) {
            adj.get(i).remove(Integer.valueOf(target));
        }
        adj.get(target).clear();

        int[] leaf = new int[n];
        rec(root, -1, leaf);

        sb.append(leaf[root]);
    }

    private static void rec(int node, int prev, int[] leaf) {
        List<Integer> childs = adj.get(node);
        if(childs.isEmpty()) { // 자식이 없다면 리프노드
            leaf[node] = 1;
            return;
        }

        for(int child : childs) {
            if(child == prev) continue;

            rec(child, node, leaf);

            leaf[node] += leaf[child]; // 현재 노드 하위에 리프 노드 갯수를 누적
        }
    }

    private static void output() throws IOException {
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));
        bw.write(sb.toString());
        bw.flush();
        bw.close();
    }
}
```
